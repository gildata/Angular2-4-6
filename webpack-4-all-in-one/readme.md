# webpack-4-all-in-one


```sh

$ mkdir webpack-4-all-in-one && cd $_

$ npm i -S webpack

$ npm i -D webpack-cli


$ npm i -D babel-core babel-loader babel-preset-env


$ npm i -S react react-dom

$ npm i -D babel-preset-react


$ npm i -D html-webpack-plugin html-loader
$ npm i -D uglifyjs-webpack-plugin copy-webpack-plugin clean-webpack-plugin

# utils
$ npm i -D rimraf browser-sync shelljs


$ npm i -D extract-text-webpack-plugin mini-css-extract-plugin css-loader sass-loader node-sass style-loader url-loader file-loader

# polyfill
$ npm i -S whatwg-fetch  babel-polyfill

$ npm i -D babel-plugin-transform-es2015-destructuring babel-plugin-transform-es2015-shorthand-properties
$ npm i -D abel-plugin-transform-object-rest-spread babel-plugin-transform-es2015-spread

$ npm i -D eslint babel-eslint eslint-plugin-react

$ npm i -D webpack-dev-server

```
